def try_finally_example(filename):
    print('* file open')
    handle = open(filename, encoding='utf-8')   # OSError
                                # нҢҢмқј open мҳӨлҘҳлҠ” finally мІҳлҰ¬к°Җ лҗҳм–ҙм„ңлҠ” м•ҲлҗҳлҜҖлЎң
                                # try лё”лЎқ л°”к№Ҙм—җм„ң мӢӨн–ү
    try:
        print('* read data')
        return handle.read()    # UnicodeDecodeError
                                # мҳҲмҷё л°ңмғқмӢң try_finally_example нҳём¶ңмһҗм—җкІҢ мҳҲмҷё л°ҳнҷҳ
    finally:
        print('* close file')
        handle.close()          # try лё”лЎқмқҙ мӢӨн–үлҗң лӢӨмқҢм—җлҠ” н•ӯмғҒ мқҙ лё”лЎқмқҙ мӢӨн–үлҗЁ
                                # => мҳҲмҷёк°Җ л°ңмғқн•ҳлҚ”лқјлҸ„, лЁјм Җ нҢҢмқјмқ„ лӢ«лҠ”лӢӨ (к·ё нӣ„м—җ мҳҲмҷё л°ҳнҷҳ)

import json

def load_json_key(data, key):
    try:
        print('* read JSON data')
        result_dict = json.loads(data)      # ValueError
    except ValueError as e:
        print('* ValueError exception')
        raise KeyError(key) from e
    else:                                   # try лё”лЎқ мҷёл¶Җ
        print('* search key')
        return result_dict[key]             # KeyError

from contextlib import contextmanager

@contextmanager
def debug_logging(level):
    logger = logging.getLogger()
    old_level = logger.getEffectiveLevel()
    logger.setLevel(level)
    try:
        # yield мӢқмқҖ with лё”лЎқмқҳ лӮҙмҡ©мқҙ мӢӨн–үлҗҳлҠ” л¶Җл¶„мқ„ мқҳлҜён•Ё
        yield
    finally:
        # with лё”лЎқмқҙ лҒқлӮҳл©ҙ мӣҗлһҳ log level лЎң ліөкө¬н•ңлӢӨ
        logger.setLevel(old_level)


with debug_logging(logging.DEBUG):
    print('* лӮҙл¶Җ лё”лЎқ')
    my_function()           # лӘЁл‘җ м¶ңл Ҙ

print('* мҷёл¶Җ лё”лЎқ')
my_function()               # error л§Ң м¶ңл Ҙ

@contextmanager
def log_level(level, name):
    logger = logging.getLogger(name)
    old_level = logger.getEffectiveLevel()
    logger.setLevel(level)
    try:
        yield logger    # as л¬ёмңјлЎң л°ӣм•„м„ң мӮ¬мҡ©н•  мҲҳ мһҲмқҢ
    finally:
        logger.setLevel(old_level)

def find_closest(sequence, goal):
    for index, value in enumerate(sequence):
        if goal < value:
            return index
    raise ValueError(f'лІ”мң„лҘј лІ—м–ҙлӮЁ: {goal}')
